    Create Store #

    Problem #

    How can I create a store?

    Solution #

    Create an in-memory store:

    package kbase

import (
	"github.com/cayleygraph/cayley"
	"log"
)

// Create a mem-store for testing
func InitStore() *cayley.Handle {
	store, err := cayley.NewMemoryGraph()
	if err != nil {
		log.Fatalln(err)
	}

	return store
}
